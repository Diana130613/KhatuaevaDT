### –ö–ª—é—á–µ–≤—ã–µ —Ñ—Ä–∞–≥–º–µ–Ω—Ç—ã –∫–æ–¥–∞
####  –†–µ–∞–ª–∏–∑–∞—Ü–∏—è —Ä–µ–∫—É—Ä—Å–∏–≤–Ω—ã—Ö –∞–ª–≥–æ—Ä–∏—Ç–º–æ–≤
```python
# recursion.py
import os
from typing import List, Optional


# 1. –ë–∏–Ω–∞—Ä–Ω—ã–π –ø–æ–∏—Å–∫ —Å —Ä–µ–∫—É—Ä—Å–∏–µ–π
def binary_search_recursive(arr: List[int], target: int,
                            left: int = 0, right: int = None) -> Optional[int]:
    """
    –†–µ–∫—É—Ä—Å–∏–≤–Ω–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è –±–∏–Ω–∞—Ä–Ω–æ–≥–æ –ø–æ–∏—Å–∫–∞

    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(log n)
    –ì–ª—É–±–∏–Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–∏: O(log n)

    Args:
        arr: –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫ —Ü–µ–ª—ã—Ö —á–∏—Å–µ–ª
        target: –∏—Å–∫–æ–º—ã–π —ç–ª–µ–º–µ–Ω—Ç
        left: –ª–µ–≤–∞—è –≥—Ä–∞–Ω–∏—Ü–∞ –ø–æ–∏—Å–∫–∞
        right: –ø—Ä–∞–≤–∞—è –≥—Ä–∞–Ω–∏—Ü–∞ –ø–æ–∏—Å–∫–∞

    Return:
        –ò–Ω–¥–µ–∫—Å —ç–ª–µ–º–µ–Ω—Ç–∞ –∏–ª–∏ None, –µ—Å–ª–∏ —ç–ª–µ–º–µ–Ω—Ç –Ω–µ –Ω–∞–π–¥–µ–Ω
    """
    if right is None:
        right = len(arr) - 1

    # –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: —ç–ª–µ–º–µ–Ω—Ç –Ω–µ –Ω–∞–π–¥–µ–Ω
    if left > right:
        return None

    # –í—ã—á–∏—Å–ª—è–µ–º —Å–µ—Ä–µ–¥–∏–Ω—É
    mid = (left + right) // 2

    # –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: —ç–ª–µ–º–µ–Ω—Ç –Ω–∞–π–¥–µ–Ω
    if arr[mid] == target:
        return mid

    # –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π —à–∞–≥: –∏—â–µ–º –≤ –ª–µ–≤–æ–π –∏–ª–∏ –ø—Ä–∞–≤–æ–π –ø–æ–ª–æ–≤–∏–Ω–µ
    elif arr[mid] > target:
        return binary_search_recursive(arr, target, left, mid - 1)
    else:
        return binary_search_recursive(arr, target, mid + 1, right)


# 2. –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –æ–±—Ö–æ–¥ —Ñ–∞–π–ª–æ–≤–æ–π —Å–∏—Å—Ç–µ–º—ã
def traverse_filesystem(path: str, indent: int = 0, max_depth: int = None,
                        current_depth: int = 0) -> None:
    """
    –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –æ–±—Ö–æ–¥ —Ñ–∞–π–ª–æ–≤–æ–π —Å–∏—Å—Ç–µ–º—ã —Å –≤—ã–≤–æ–¥–æ–º –¥–µ—Ä–µ–≤–∞ –∫–∞—Ç–∞–ª–æ–≥–æ–≤

    Args:
        path: –Ω–∞—á–∞–ª—å–Ω—ã–π –ø—É—Ç—å –¥–ª—è –æ–±—Ö–æ–¥–∞
        indent: –æ—Ç—Å—Ç—É–ø –¥–ª—è –≤–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏–∏ –∏–µ—Ä–∞—Ä—Ö–∏–∏
        max_depth: –º–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –≥–ª—É–±–∏–Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–∏ (–¥–ª—è –∏—Å—Å–ª–µ–¥–æ–≤–∞–Ω–∏—è)
        current_depth: —Ç–µ–∫—É—â–∞—è –≥–ª—É–±–∏–Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–∏
    """
    if max_depth is not None and current_depth > max_depth:
        return

    try:
        # –ü–æ–ª—É—á–∞–µ–º —Å–ø–∏—Å–æ–∫ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –≤ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏
        items = os.listdir(path)
    except PermissionError:
        print(" " * indent + "[–î–æ—Å—Ç—É–ø –∑–∞–ø—Ä–µ—â–µ–Ω]")
        return
    except FileNotFoundError:
        print(" " * indent + "[–ü—É—Ç—å –Ω–µ –Ω–∞–π–¥–µ–Ω]")
        return

    for item in sorted(items):
        item_path = os.path.join(path, item)

        if os.path.isdir(item_path):
            # –≠—Ç–æ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—è - –≤—ã–≤–æ–¥–∏–º –∏ —Ä–µ–∫—É—Ä—Å–∏–≤–Ω–æ –æ–±—Ö–æ–¥–∏–º
            print(" " * indent + f" {item}/")
            traverse_filesystem(item_path, indent + 2, max_depth, current_depth + 1)
        else:
            # –≠—Ç–æ —Ñ–∞–π–ª - –ø—Ä–æ—Å—Ç–æ –≤—ã–≤–æ–¥–∏–º
            print(" " * indent + f" {item}")


# 3. –•–∞–Ω–æ–π—Å–∫–∏–µ –±–∞—à–Ω–∏
def hanoi_towers(n: int, source: str = "A",
                 auxiliary: str = "B", target: str = "C") -> List[str]:
    """
    –†–µ—à–µ–Ω–∏–µ –∑–∞–¥–∞—á–∏ "–•–∞–Ω–æ–π—Å–∫–∏–µ –±–∞—à–Ω–∏"

    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(2^n)
    –ì–ª—É–±–∏–Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–∏: O(n)

    Args:
        n: –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –¥–∏—Å–∫–æ–≤
        source: –Ω–∞—á–∞–ª—å–Ω—ã–π —Å—Ç–µ—Ä–∂–µ–Ω—å
        auxiliary: –≤—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω—ã–π —Å—Ç–µ—Ä–∂–µ–Ω—å
        target: —Ü–µ–ª–µ–≤–æ–π —Å—Ç–µ—Ä–∂–µ–Ω—å

    Returns:
        –°–ø–∏—Å–æ–∫ —Ö–æ–¥–æ–≤ –¥–ª—è —Ä–µ—à–µ–Ω–∏—è –∑–∞–¥–∞—á–∏
    """
    moves = []

    def _hanoi(n: int, source: str, auxiliary: str, target: str):
        # –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: –ø–µ—Ä–µ–º–µ—â–∞–µ–º –æ–¥–∏–Ω –¥–∏—Å–∫
        if n == 1:
            moves.append(f"–ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å –¥–∏—Å–∫ 1 —Å {source} –Ω–∞ {target}")
            return

        # –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π —à–∞–≥:
        # 1. –ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å n-1 –¥–∏—Å–∫–æ–≤ –Ω–∞ –≤—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω—ã–π —Å—Ç–µ—Ä–∂–µ–Ω—å
        _hanoi(n - 1, source, target, auxiliary)

        # 2. –ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å —Å–∞–º—ã–π –±–æ–ª—å—à–æ–π –¥–∏—Å–∫ –Ω–∞ —Ü–µ–ª–µ–≤–æ–π —Å—Ç–µ—Ä–∂–µ–Ω—å
        moves.append(f"–ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å –¥–∏—Å–∫ {n} —Å {source} –Ω–∞ {target}")

        # 3. –ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å n-1 –¥–∏—Å–∫–æ–≤ —Å –≤—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω–æ–≥–æ –Ω–∞ —Ü–µ–ª–µ–≤–æ–π —Å—Ç–µ—Ä–∂–µ–Ω—å
        _hanoi(n - 1, auxiliary, source, target)

    _hanoi(n, source, auxiliary, target)
    return moves


# –§—É–Ω–∫—Ü–∏–∏ –¥–ª—è —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–æ–≤
def measure_max_recursion_depth(start_path: str = ".") -> int:
    """
    –ò–∑–º–µ—Ä–µ–Ω–∏–µ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–π –≥–ª—É–±–∏–Ω—ã —Ä–µ–∫—É—Ä—Å–∏–∏ –ø—Ä–∏ –æ–±—Ö–æ–¥–µ —Ñ–∞–π–ª–æ–≤–æ–π —Å–∏—Å—Ç–µ–º—ã
    """
    max_depth = [0]

    def _traverse_measure(path: str, current_depth: int):
        max_depth[0] = max(max_depth[0], current_depth)

        try:
            for item in os.listdir(path):
                item_path = os.path.join(path, item)
                if os.path.isdir(item_path):
                    _traverse_measure(item_path, current_depth + 1)
        except (PermissionError, FileNotFoundError):
            pass

    _traverse_measure(start_path, 0)
    return max_depth[0]
```

```python
# recursion_tasks.py
import os
from typing import List, Optional


# 1. –ë–∏–Ω–∞—Ä–Ω—ã–π –ø–æ–∏—Å–∫ —Å —Ä–µ–∫—É—Ä—Å–∏–µ–π
def binary_search_recursive(arr: List[int], target: int,
                            left: int = 0, right: int = None) -> Optional[int]:
    """
    –†–µ–∫—É—Ä—Å–∏–≤–Ω–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è –±–∏–Ω–∞—Ä–Ω–æ–≥–æ –ø–æ–∏—Å–∫–∞

    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(log n)
    –ì–ª—É–±–∏–Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–∏: O(log n)

    Args:
        arr: –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫ —Ü–µ–ª—ã—Ö —á–∏—Å–µ–ª
        target: –∏—Å–∫–æ–º—ã–π —ç–ª–µ–º–µ–Ω—Ç
        left: –ª–µ–≤–∞—è –≥—Ä–∞–Ω–∏—Ü–∞ –ø–æ–∏—Å–∫–∞
        right: –ø—Ä–∞–≤–∞—è –≥—Ä–∞–Ω–∏—Ü–∞ –ø–æ–∏—Å–∫–∞

    Return:
        –ò–Ω–¥–µ–∫—Å —ç–ª–µ–º–µ–Ω—Ç–∞ –∏–ª–∏ None, –µ—Å–ª–∏ —ç–ª–µ–º–µ–Ω—Ç –Ω–µ –Ω–∞–π–¥–µ–Ω
    """
    if right is None:
        right = len(arr) - 1

    # –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: —ç–ª–µ–º–µ–Ω—Ç –Ω–µ –Ω–∞–π–¥–µ–Ω
    if left > right:
        return None

    # –í—ã—á–∏—Å–ª—è–µ–º —Å–µ—Ä–µ–¥–∏–Ω—É
    mid = (left + right) // 2

    # –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: —ç–ª–µ–º–µ–Ω—Ç –Ω–∞–π–¥–µ–Ω
    if arr[mid] == target:
        return mid

    # –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π —à–∞–≥: –∏—â–µ–º –≤ –ª–µ–≤–æ–π –∏–ª–∏ –ø—Ä–∞–≤–æ–π –ø–æ–ª–æ–≤–∏–Ω–µ
    elif arr[mid] > target:
        return binary_search_recursive(arr, target, left, mid - 1)
    else:
        return binary_search_recursive(arr, target, mid + 1, right)


# 2. –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –æ–±—Ö–æ–¥ —Ñ–∞–π–ª–æ–≤–æ–π —Å–∏—Å—Ç–µ–º—ã
def traverse_filesystem(path: str, indent: int = 0, max_depth: int = None,
                        current_depth: int = 0) -> None:
    """
    –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –æ–±—Ö–æ–¥ —Ñ–∞–π–ª–æ–≤–æ–π —Å–∏—Å—Ç–µ–º—ã —Å –≤—ã–≤–æ–¥–æ–º –¥–µ—Ä–µ–≤–∞ –∫–∞—Ç–∞–ª–æ–≥–æ–≤

    Args:
        path: –Ω–∞—á–∞–ª—å–Ω—ã–π –ø—É—Ç—å –¥–ª—è –æ–±—Ö–æ–¥–∞
        indent: –æ—Ç—Å—Ç—É–ø –¥–ª—è –≤–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏–∏ –∏–µ—Ä–∞—Ä—Ö–∏–∏
        max_depth: –º–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –≥–ª—É–±–∏–Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–∏ (–¥–ª—è –∏—Å—Å–ª–µ–¥–æ–≤–∞–Ω–∏—è)
        current_depth: —Ç–µ–∫—É—â–∞—è –≥–ª—É–±–∏–Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–∏
    """
    if max_depth is not None and current_depth > max_depth:
        return

    try:
        # –ü–æ–ª—É—á–∞–µ–º —Å–ø–∏—Å–æ–∫ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –≤ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏
        items = os.listdir(path)
    except PermissionError:
        print(" " * indent + "[–î–æ—Å—Ç—É–ø –∑–∞–ø—Ä–µ—â–µ–Ω]")
        return
    except FileNotFoundError:
        print(" " * indent + "[–ü—É—Ç—å –Ω–µ –Ω–∞–π–¥–µ–Ω]")
        return

    for item in sorted(items):
        item_path = os.path.join(path, item)

        if os.path.isdir(item_path):
            # –≠—Ç–æ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—è - –≤—ã–≤–æ–¥–∏–º –∏ —Ä–µ–∫—É—Ä—Å–∏–≤–Ω–æ –æ–±—Ö–æ–¥–∏–º
            print(" " * indent + f"üìÅ {item}/")
            traverse_filesystem(item_path, indent + 2, max_depth, current_depth + 1)
        else:
            # –≠—Ç–æ —Ñ–∞–π–ª - –ø—Ä–æ—Å—Ç–æ –≤—ã–≤–æ–¥–∏–º
            print(" " * indent + f"üìÑ {item}")


# 3. –•–∞–Ω–æ–π—Å–∫–∏–µ –±–∞—à–Ω–∏
def hanoi_towers(n: int, source: str = "A",
                 auxiliary: str = "B", target: str = "C") -> List[str]:
    """
    –†–µ—à–µ–Ω–∏–µ –∑–∞–¥–∞—á–∏ "–•–∞–Ω–æ–π—Å–∫–∏–µ –±–∞—à–Ω–∏"

    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(2^n)
    –ì–ª—É–±–∏–Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–∏: O(n)

    Args:
        n: –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –¥–∏—Å–∫–æ–≤
        source: –Ω–∞—á–∞–ª—å–Ω—ã–π —Å—Ç–µ—Ä–∂–µ–Ω—å
        auxiliary: –≤—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω—ã–π —Å—Ç–µ—Ä–∂–µ–Ω—å
        target: —Ü–µ–ª–µ–≤–æ–π —Å—Ç–µ—Ä–∂–µ–Ω—å

    Returns:
        –°–ø–∏—Å–æ–∫ —Ö–æ–¥–æ–≤ –¥–ª—è —Ä–µ—à–µ–Ω–∏—è –∑–∞–¥–∞—á–∏
    """
    moves = []

    def _hanoi(n: int, source: str, auxiliary: str, target: str):
        # –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: –ø–µ—Ä–µ–º–µ—â–∞–µ–º –æ–¥–∏–Ω –¥–∏—Å–∫
        if n == 1:
            moves.append(f"–ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å –¥–∏—Å–∫ 1 —Å {source} –Ω–∞ {target}")
            return

        # –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π —à–∞–≥:
        # 1. –ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å n-1 –¥–∏—Å–∫–æ–≤ –Ω–∞ –≤—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω—ã–π —Å—Ç–µ—Ä–∂–µ–Ω—å
        _hanoi(n - 1, source, target, auxiliary)

        # 2. –ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å —Å–∞–º—ã–π –±–æ–ª—å—à–æ–π –¥–∏—Å–∫ –Ω–∞ —Ü–µ–ª–µ–≤–æ–π —Å—Ç–µ—Ä–∂–µ–Ω—å
        moves.append(f"–ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å –¥–∏—Å–∫ {n} —Å {source} –Ω–∞ {target}")

        # 3. –ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å n-1 –¥–∏—Å–∫–æ–≤ —Å –≤—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω–æ–≥–æ –Ω–∞ —Ü–µ–ª–µ–≤–æ–π —Å—Ç–µ—Ä–∂–µ–Ω—å
        _hanoi(n - 1, auxiliary, source, target)

    _hanoi(n, source, auxiliary, target)
    return moves


# –§—É–Ω–∫—Ü–∏–∏ –¥–ª—è —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–æ–≤
def measure_max_recursion_depth(start_path: str = ".") -> int:
    """
    –ò–∑–º–µ—Ä–µ–Ω–∏–µ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–π –≥–ª—É–±–∏–Ω—ã —Ä–µ–∫—É—Ä—Å–∏–∏ –ø—Ä–∏ –æ–±—Ö–æ–¥–µ —Ñ–∞–π–ª–æ–≤–æ–π —Å–∏—Å—Ç–µ–º—ã
    """
    max_depth = [0]

    def _traverse_measure(path: str, current_depth: int):
        max_depth[0] = max(max_depth[0], current_depth)

        try:
            for item in os.listdir(path):
                item_path = os.path.join(path, item)
                if os.path.isdir(item_path):
                    _traverse_measure(item_path, current_depth + 1)
        except (PermissionError, FileNotFoundError):
            pass

    _traverse_measure(start_path, 0)
    return max_depth[0]
```
